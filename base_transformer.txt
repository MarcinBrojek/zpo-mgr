from lark import Token, Transformer
from classes import Var

class OptimusPirme(Transformer):
    def var(self, c):
        return Var(c[0], c[1])

    def PY(self, c):
        return str(c[1:-1])

    def ID(self, c):
        return str(c)

    def STR(self, c):
        return str(c)

    def I(self, c):
        return int(c)
